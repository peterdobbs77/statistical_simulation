generate two uniforms for g(mu,sig2)=g(mu)g(sig2)
m=10^6;

% uniform for g(sig2) pick wide range
asig2=0; bsig2=50;
usig2=(bsig2-asig2)*rand(m,1)+asig2;
figure;
histogram(usig2)
% use bivariate (independent) uniform random numbers
% q(mu,sig2)=h(mu,sig2)*f(mu,sig2)/g(mu,sig2)
% g(mu,sig2) is bivariate uniform
gmusig2=1/((bmu-amu).*(bsig2-asig2));
fmusig2=zeros(m,1);
for i=1:m
% note here that not using normalizing constant f(y)
% but determining it by importance sampling
% then dividing by it when calculating estimates
% way using some math for nornal, normal-inv gamma
% qmusig2(i,1)=(2*pi)^(-(n+1)/2)*sqrt(alp)...
% *kap^((nu-2)/2)/gamma((nu-2)/2)/(2^((nu-2)/2))...
% *(usig2(i,1))^(-(n+nu+1)/2)*exp(-(...
% sum((y-umu(i,1)).^2)...
% +alp*(umu(i,1)-mu0)^2+kap)/2/usig2(i,1));
% here explicitly just multilying likelihood and priors, no f(y)
% calculating f(y) later and dividing to normalize
fmusig2(i,1)=...
(2*pi*usig2(i,1))^(-n/2)*exp(-sum((y-umu(i,1)).^2)/2/usig2(i,1))...
*(2*pi*usig2(i,1)/alp)^(-1/2)*exp(-(umu(i,1)-mu0)^2/2/usig2(i,1)*alp)...
*(kap/2)^((nu-2)/2)/gamma((nu-2)/2)*(usig2(i,1))^(-nu/2)*exp(-kap/2/usig2(i,1));
end
% determine normalizing constant
fyIS =mean(1*fmusig2/gmusig2);
% determine expected value of mu and expected variance of mu
muIS =mean(umu .*(fmusig2/fyIS)/gmusig2);

varmuIS =mu2IS-muIS^2;
